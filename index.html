int solve(vector<int>& arr) {
  int n = arr.size();
  int ans = 0;

  // O(n log n) approach
  sort(arr.begin(), arr.end());

  for (int i = 0; i < n; i++) {
    ans = max(ans, arr[i] * (n - i));
  }

  return ans;
}

// Time: O(n log n)
// Space: O(1) ✓
